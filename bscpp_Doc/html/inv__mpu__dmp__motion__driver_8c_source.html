722int dmp_set_tap_thresh(unsigned char axis, unsigned short thresh)
723{
724 unsigned char tmp[4], accel_fsr;
725 float scaled_thresh;
726 unsigned short dmp_thresh, dmp_thresh_2;
727 if (!(axis & TAP_XYZ) || thresh > 1600)
728 return -1;
729
730 scaled_thresh = (float)thresh / DMP_SAMPLE_RATE;
731
732 mpu_get_accel_fsr(&accel_fsr);
733 switch (accel_fsr) {
734 case 2:
735 dmp_thresh = (unsigned short)(scaled_thresh * 16384);
736 /* dmp_thresh * 0.75 */
737 dmp_thresh_2 = (unsigned short)(scaled_thresh * 12288);
738 break;
739 case 4:
740 dmp_thresh = (unsigned short)(scaled_thresh * 8192);
741 /* dmp_thresh * 0.75 */
742 dmp_thresh_2 = (unsigned short)(scaled_thresh * 6144);
743 break;
744 case 8:
745 dmp_thresh = (unsigned short)(scaled_thresh * 4096);
746 /* dmp_thresh * 0.75 */

748 break;
749 case 16:
750 dmp_thresh = (unsigned short)(scaled_thresh * 2048);
751 /* dmp_thresh * 0.75 */
752 dmp_thresh_2 = (unsigned short)(scaled_thresh * 1536);
753 break;
754 default:
755 return -1;
756 }
757 tmp[0] = (unsigned char)(dmp_thresh >> 8);
758 tmp[1] = (unsigned char)(dmp_thresh & 0xFF);
759 tmp[2] = (unsigned char)(dmp_thresh_2 >> 8);
760 tmp[3] = (unsigned char)(dmp_thresh_2 & 0xFF);
761
762 if (axis & TAP_X) {
763 if (mpu_write_mem(DMP_TAP_THX, 2, tmp))

776 return -1;
777 if (mpu_write_mem(D_1_44, 2, tmp+2))
778 return -1;

780 return 0;

997int dmp_enable_feature(unsigned short mask)
998{
999 unsigned char tmp[10];
1000
1001 /* TODO: All of these settings can probably be integrated into the default
1002 * DMP image.
1003 */
1004 /* Set integration scale factor. */
1005 tmp[0] = (unsigned char)((GYRO_SF >> 24) & 0xFF);
1006 tmp[1] = (unsigned char)((GYRO_SF >> 16) & 0xFF);
1007 tmp[2] = (unsigned char)((GYRO_SF >> 8) & 0xFF);
1008 tmp[3] = (unsigned char)(GYRO_SF & 0xFF);
1009 mpu_write_mem(D_0_104, 4, tmp);
1010
1011 /* Send sensor data to the FIFO. */
1012 tmp[0] = 0xA3;
1013 if (mask & DMP_FEATURE_SEND_RAW_ACCEL) {
1014 tmp[1] = 0xC0;
1015 tmp[2] = 0xC8;
1016 tmp[3] = 0xC2;
1017 } else {
1018 tmp[1] = 0xA3;
1019 tmp[2] = 0xA3;
1020 tmp[3] = 0xA3;
1021 }
1022 if (mask & DMP_FEATURE_SEND_ANY_GYRO) {
1023 tmp[4] = 0xC4;
1024 tmp[5] = 0xCC;
1025 tmp[6] = 0xC6;
1026 } else {
1027 tmp[4] = 0xA3;
1028 tmp[5] = 0xA3;
1029 tmp[6] = 0xA3;
1030 }
1031 tmp[7] = 0xA3;
1032 tmp[8] = 0xA3;
1033 tmp[9] = 0xA3;
1034 mpu_write_mem(CFG_15,10,tmp);
1035
1036 /* Send gesture data to the FIFO. */
1037 if (mask & (DMP_FEATURE_TAP | DMP_FEATURE_ANDROID_ORIENT))
1038 tmp[0] = DINA20;
1039 else
1040 tmp[0] = 0xD8;
1041 mpu_write_mem(CFG_27,1,tmp);
1042
1043 if (mask & DMP_FEATURE_GYRO_CAL)
1044 dmp_enable_gyro_cal(1);
1045 else
1046 dmp_enable_gyro_cal(0);
1047
1048 if (mask & DMP_FEATURE_SEND_ANY_GYRO) {
1049 if (mask & DMP_FEATURE_SEND_CAL_GYRO) {
1050 tmp[0] = 0xB2;
1051 tmp[1] = 0x8B;
1052 tmp[2] = 0xB6;
1053 tmp[3] = 0x9B;
1054 } else {
1055 tmp[0] = DINAC0;
1056 tmp[1] = DINA80;
1057 tmp[2] = DINAC2;
1058 tmp[3] = DINA90;
1059 }
1060 mpu_write_mem(CFG_GYRO_RAW_DATA, 4, tmp);
1061 }
1062
1063 if (mask & DMP_FEATURE_TAP) {
1064 /* Enable tap. */
1065 tmp[0] = 0xF8;
1066 mpu_write_mem(CFG_20, 1, tmp);
1067 dmp_set_tap_thresh(TAP_XYZ, 250);
1068 dmp_set_tap_axes(TAP_XYZ);
1069 dmp_set_tap_count(1);
1070 dmp_set_tap_time(100);
1071 dmp_set_tap_time_multi(500);
1072
1073 dmp_set_shake_reject_thresh(GYRO_SF, 200);
1074 dmp_set_shake_reject_time(40);
1075 dmp_set_shake_reject_timeout(10);
1076 } else {
1077 tmp[0] = 0xD8;
1078 mpu_write_mem(CFG_20, 1, tmp);
1079 }
1080
1081 if (mask & DMP_FEATURE_ANDROID_ORIENT) {
1082 tmp[0] = 0xD9;
1083 } else
1084 tmp[0] = 0xD8;
1085 mpu_write_mem(CFG_ANDROID_ORIENT_INT, 1, tmp);
1086
1087 if (mask & DMP_FEATURE_LP_QUAT)
1088 dmp_enable_lp_quat(1);
1089 else
1090 dmp_enable_lp_quat(0);
1091
1092 if (mask & DMP_FEATURE_6X_LP_QUAT)
1093 dmp_enable_6x_lp_quat(1);
1094 else
1095 dmp_enable_6x_lp_quat(0);
1096
1097 /* Pedometer is always enabled. */
1098 dmp.feature_mask = mask | DMP_FEATURE_PEDOMETER;
1099 mpu_reset_fifo();
1100
1101 dmp.packet_length = 0;
1102 if (mask & DMP_FEATURE_SEND_RAW_ACCEL)
1103 dmp.packet_length += 6;
1104 if (mask & DMP_FEATURE_SEND_ANY_GYRO)
1105 dmp.packet_length += 6;
1106 if (mask & (DMP_FEATURE_LP_QUAT | DMP_FEATURE_6X_LP_QUAT))
1107 dmp.packet_length += 16;
1108 if (mask & (DMP_FEATURE_TAP | DMP_FEATURE_ANDROID_ORIENT))
1109 dmp.packet_length += 4;
1110
1111 return 0;
1112}

1268int dmp_read_fifo(short *gyro, short *accel, long *quat,
1269 unsigned long *timestamp, short *sensors, unsigned char *more)
1270{
1271 unsigned char fifo_data[MAX_PACKET_LENGTH];
1272 unsigned char ii = 0;
1273
1274 /* TODO: sensors[0] only changes when dmp_enable_feature is called. We can
1275 * cache this value and save some cycles.
1276 */
1277 sensors[0] = 0;
1278
1279 /* Get a packet. */
1280 if (mpu_read_fifo_stream(dmp.packet_length, fifo_data, more))
1281 return -1;
1282
1283 /* Parse DMP packet. */
1284 if (dmp.feature_mask & (DMP_FEATURE_LP_QUAT | DMP_FEATURE_6X_LP_QUAT)) {
1285#ifdef FIFO_CORRUPTION_CHECK
1286 long quat_q14[4], quat_mag_sq;
1287#endif
1288 quat[0] = ((long)fifo_data[0] << 24) | ((long)fifo_data[1] << 16) |
1289 ((long)fifo_data[2] << 8) | fifo_data[3];
1290 quat[1] = ((long)fifo_data[4] << 24) | ((long)fifo_data[5] << 16) |
1291 ((long)fifo_data[6] << 8) | fifo_data[7];
1292 quat[2] = ((long)fifo_data[8] << 24) | ((long)fifo_data[9] << 16) |
1293 ((long)fifo_data[10] << 8) | fifo_data[11];
1294 quat[3] = ((long)fifo_data[12] << 24) | ((long)fifo_data[13] << 16) |
1295 ((long)fifo_data[14] << 8) | fifo_data[15];
1296 ii += 16;
1297#ifdef FIFO_CORRUPTION_CHECK
1298 /* We can detect a corrupted FIFO by monitoring the quaternion data and
1299 * ensuring that the magnitude is always normalized to one. This
1300 * shouldn't happen in normal operation, but if an I2C error occurs,
1301 * the FIFO reads might become misaligned.
1302 *
1303 * Let's start by scaling down the quaternion data to avoid long long
1304 * math.
1305 */
1306 quat_q14[0] = quat[0] >> 16;
1307 quat_q14[1] = quat[1] >> 16;
1308 quat_q14[2] = quat[2] >> 16;
1309 quat_q14[3] = quat[3] >> 16;
1310 quat_mag_sq = quat_q14[0] * quat_q14[0] + quat_q14[1] * quat_q14[1] +
1311 quat_q14[2] * quat_q14[2] + quat_q14[3] * quat_q14[3];
1312 if ((quat_mag_sq < QUAT_MAG_SQ_MIN) ||
1313 (quat_mag_sq > QUAT_MAG_SQ_MAX)) {
1314 /* Quaternion is outside of the acceptable threshold. */
1315 mpu_reset_fifo();
1316 sensors[0] = 0;
1317 return -1;
1318 }
1319 sensors[0] |= INV_WXYZ_QUAT;
1320#endif
1321 }
1322
1323 if (dmp.feature_mask & DMP_FEATURE_SEND_RAW_ACCEL) {
1324 accel[0] = ((short)fifo_data[ii+0] << 8) | fifo_data[ii+1];
1325 accel[1] = ((short)fifo_data[ii+2] << 8) | fifo_data[ii+3];
1326 accel[2] = ((short)fifo_data[ii+4] << 8) | fifo_data[ii+5];
1327 ii += 6;
1328 sensors[0] |= INV_XYZ_ACCEL;
1329 }
1330
1331 if (dmp.feature_mask & DMP_FEATURE_SEND_ANY_GYRO) {
1332 gyro[0] = ((short)fifo_data[ii+0] << 8) | fifo_data[ii+1];
1333 gyro[1] = ((short)fifo_data[ii+2] << 8) | fifo_data[ii+3];
1334 gyro[2] = ((short)fifo_data[ii+4] << 8) | fifo_data[ii+5];
1335 ii += 6;
1336 sensors[0] |= INV_XYZ_GYRO;
1337 }
1338
1339 /* Gesture data is at the end of the DMP packet. Parse it and call
1340 * the gesture callbacks (if registered).
1341 */
1342 if (dmp.feature_mask & (DMP_FEATURE_TAP | DMP_FEATURE_ANDROID_ORIENT))
1343 decode_gesture(fifo_data + ii);
1344
1345 get_ms(timestamp);
1346 return 0;
1347}
