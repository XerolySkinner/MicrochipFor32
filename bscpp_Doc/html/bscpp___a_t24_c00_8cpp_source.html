72void bscpp_STM32_AT24C00::WriteByte(u8 addr,u8 *dat,u8 len){
73 unsigned int member=0;
74 IIC_Start();
75 IIC_SendByte(Address<<1);
76 IIC_WaitAck();
77 IIC_SendByte(addr);
78 IIC_WaitAck();
79 do{
80 IIC_SendByte(dat[member]);
81 IIC_WaitAck();
82 member++;
83 if(((member+addr)%Page)==0){
84 IIC_Stop();
85 IIC_Delay();
86 IIC_Start();
87 IIC_SendByte(Address<<1);
88 IIC_WaitAck();
89 IIC_SendByte(addr+member);
90 IIC_WaitAck();}
91 }while(member!=len);
92 IIC_Stop();
93 IIC_Delay();}
