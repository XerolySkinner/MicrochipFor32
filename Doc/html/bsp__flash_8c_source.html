56uint32_t bsp_WriteFlashOnePage(uint32_t *dat,uint32_t num){
57 FLASH_EraseInitTypeDef f;
58 uint32_t PageError = 0; //错误号7
59 HAL_FLASH_Unlock(); //解锁FLASH
60 f.TypeErase = FLASH_TYPEERASE_PAGES; //Flash擦除
61 f.PageAddress = FlashAddr; //擦除地址(起始页)
62 f.NbPages = 1; //擦除1页
63 HAL_FLASHEx_Erase(&f,&PageError); //调用擦除函数
64 for(int i=0;i<num;i++)
65 HAL_FLASH_Program(TYPEPROGRAM_WORD,FlashAddr+i*4,dat[i]); //对FLASH烧写
66 HAL_FLASH_Lock(); //锁住FLASH
67 return PageError;}
