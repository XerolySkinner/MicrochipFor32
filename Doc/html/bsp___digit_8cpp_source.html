41void bsp_Digit::dat(u8 SendVal) {
42 for (u8 i = 0; i < 8; i++) {
43 HAL_GPIO_WritePin(GPIOx_MOSIO,GPIO_Pin_MOSIO,((SendVal << i) & 0x80)?GPIO_PIN_SET: GPIO_PIN_RESET);
44 HAL_GPIO_WritePin(GPIOx_S_CLK,GPIO_Pin_S_CLK,GPIO_PIN_RESET);
45 HAL_GPIO_WritePin(GPIOx_S_CLK, GPIO_Pin_S_CLK, GPIO_PIN_SET);}}

51void bsp_Digit::DatCom(u8 SendVal,u8 cha){
52 u16 i=0xFFFF/division;
53 dat(Polarity?~digit_ca[SendVal]:digit_ca[SendVal]);
54 dat(1<<cha);
55 com();
56 while(++i);}
57//----------------------------------------------------------------------------------------------------
58void bsp_Digit::num(u16 num){
59 DatCom(num%10,0);num/=10;
60 DatCom(num%10,1);num/=10;
61 DatCom(num%10,2);num/=10;
62 DatCom(num%10,3);num/=10;}

64void bsp_Digit::dop(u8 cha){
65 u16 i=0xFFFF/division;
66 dat(Polarity?0x80:0x7E);
67 dat(1<<cha);
68 com();
69 while(++i);}
