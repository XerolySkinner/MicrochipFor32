84void bsp_STM32_OLED::Clear(u8 dat) {
85 u8 page;
86 for (page = 0; page < 8; page++) {
87 Page_set(page);
88 Column_set(0);
89 IIC_Start();
90 IIC_SendByte(0x3C << 1);
91 IIC_WaitAck();
92 IIC_SendByte(0x40);
93 IIC_WaitAck();
94 for (int i = 0; i < 128; i++) {
95 IIC_SendByte(dat);
96 IIC_WaitAck();}
97 IIC_Stop();}}

146 word -= 33;
